function get_breadcrumbs() {
    $breadcrumbs = array();

    // Home page
    $breadcrumbs[] = array(
        'title' => 'Home',
        'url' => home_url('/'),
    );

    if (is_single()) {
        // Post category
        $categories = get_the_category();
        if ($categories) {
            $category = $categories[0];
            $breadcrumbs[] = array(
                'title' => $category->name,
                'url' => get_category_link($category->term_id),
            );
        }

        // Current post
        $breadcrumbs[] = array(
            'title' => get_the_title(),
            'url' => get_permalink(),
        );

    } elseif (is_page()) {

        // Page
        $breadcrumbs[] = array(
            'title' => get_the_title(),
            'url' => get_permalink(),
        );

    } elseif (is_category()) {

        // Category
        $breadcrumbs[] = array(
            'title' => single_cat_title('', false),
            'url' => get_category_link(get_queried_object_id()),
        );
    }

    return $breadcrumbs;
}
